# FPV Tools Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a browser-native FPV tools suite — PID simulator with live scope/quad preview, calculators (power, motor, RF), converters, blackbox analyzer — deployed to Vercel.

**Architecture:** Vue 3 Composition API shell for routing/layout. Lit 3 web components (`fpv-` prefix) for all UI primitives and tool components. Pure TypeScript core for physics/math (zero DOM, zero framework). Canvas 2D for all plots and visualizations. Vite 6 builds everything.

**Tech Stack:** Vue 3.5, Lit 3, TypeScript 5.8, Vite 6, Vitest, vue-router 4, Canvas 2D API

## Global Constraints

- Node.js 22+ (LTS)
- No CSS frameworks — design tokens as CSS custom properties only
- No runtime dependencies beyond Vue, Lit, vue-router
- All core modules: pure functions, zero DOM access, zero side effects
- Dark theme default. Light via `[data-theme="light"]` on `<html>`
- Custom element prefix: `fpv-` for primitives, `pid-`/`pack-`/`bbl-` for tool-specific
- Internal units: SI (rad/s, kg·m², N·m, seconds). deg/s only at input/display boundary
- Seeded PRNG for all randomness — deterministic output for same config
- `Float32Array` for all signal buffers
- Semi-implicit Euler integration (fixed-step, matching real flight controllers)

```ts
import { ref, watch, type Ref } from 'vue'

export function useLocalStorage<T>(key: string, defaultValue: T): Ref<T> {
  const stored = localStorage.getItem(key)
  const data = ref<T>(stored ? JSON.parse(stored) : defaultValue) as Ref<T>
  watch(data, (v) => localStorage.setItem(key, JSON.stringify(v)), { deep: true })
  return data
}
```

- [ ] **Step 2: Create src/app/composables/useTheme.ts**

```ts
import { useLocalStorage } from './useLocalStorage'
import { watchEffect } from 'vue'

export type Theme = 'dark' | 'light' | 'auto'

export function useTheme() {
  const theme = useLocalStorage<Theme>('fpv-tools:theme', 'dark')

  watchEffect(() => {
    document.documentElement.setAttribute('data-theme', theme.value)
  })

  function cycle() {
    const order: Theme[] = ['dark', 'light', 'auto']
    const i = order.indexOf(theme.value)
    theme.value = order[(i + 1) % order.length]
  }

  return { theme, cycle }
}
```

- [ ] **Step 3: Create src/app/composables/useUrlState.ts**

```ts
import { ref, watch, onMounted, type Ref } from 'vue'
import { useRoute, useRouter } from 'vue-router'

export function useUrlState<T>(key: string, defaultValue: T): Ref<T> {
  const route = useRoute()
  const router = useRouter()
  const state = ref<T>(defaultValue) as Ref<T>

  onMounted(() => {
    const param = route.query[key] as string | undefined
    if (param) {
      try { state.value = JSON.parse(atob(param)) } catch {}
    }
  })

  watch(state, (v) => {
    const encoded = btoa(JSON.stringify(v))
    router.replace({ query: { ...route.query, [key]: encoded } })
  }, { deep: true })

  return state
}
```

```ts
// src/core/shared/math.ts
export function clamp(v: number, min: number, max: number): number {
  return v < min ? min : v > max ? max : v
}

export function lerp(a: number, b: number, t: number): number {
  return a + (b - a) * t
}

export function remap(v: number, inMin: number, inMax: number, outMin: number, outMax: number): number {
  return outMin + ((v - inMin) / (inMax - inMin)) * (outMax - outMin)
}

export function rms(arr: Float32Array): number {
  let sum = 0
  for (let i = 0; i < arr.length; i++) sum += arr[i] * arr[i]
  return Math.sqrt(sum / arr.length)
}
```

```ts
// src/core/shared/prng.ts
export function mulberry32(seed: number): () => number {
  let s = seed | 0
  return () => {
    s = (s + 0x6d2b79f5) | 0
    let t = Math.imul(s ^ (s >>> 15), 1 | s)
    t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296
  }
}

export function gaussianNoise(rng: () => number): number {
  let u = 0, v = 0
  while (u === 0) u = rng()
  while (v === 0) v = rng()
  return Math.sqrt(-2.0 * Math.log(u)) * Math.cos(2.0 * Math.PI * v)
}
```

```ts
// src/core/shared/biquad.ts
export class BiquadFilter {
  private b0: number; private b1: number; private b2: number
  private a1: number; private a2: number
  private x1 = 0; private x2 = 0; private y1 = 0; private y2 = 0

  constructor(b0: number, b1: number, b2: number, a1: number, a2: number) {
    this.b0 = b0; this.b1 = b1; this.b2 = b2; this.a1 = a1; this.a2 = a2
  }

  process(x: number): number {
    const y = this.b0 * x + this.b1 * this.x1 + this.b2 * this.x2
             - this.a1 * this.y1 - this.a2 * this.y2
    this.x2 = this.x1; this.x1 = x
    this.y2 = this.y1; this.y1 = y
    return y
  }

  reset() { this.x1 = this.x2 = this.y1 = this.y2 = 0 }

  static lowpass(fc: number, fs: number, q: number): BiquadFilter {
    const w0 = 2 * Math.PI * fc / fs
    const alpha = Math.sin(w0) / (2 * q)
    const cos_w0 = Math.cos(w0)
    const a0 = 1 + alpha
    return new BiquadFilter(
      (1 - cos_w0) / 2 / a0,
      (1 - cos_w0) / a0,
      (1 - cos_w0) / 2 / a0,
      -2 * cos_w0 / a0,
      (1 - alpha) / a0,
    )
  }

  static notch(fc: number, fs: number, q: number): BiquadFilter {
    const w0 = 2 * Math.PI * fc / fs
    const alpha = Math.sin(w0) / (2 * q)
    const cos_w0 = Math.cos(w0)
    const a0 = 1 + alpha
    return new BiquadFilter(
      1 / a0,
      -2 * cos_w0 / a0,
      1 / a0,
      -2 * cos_w0 / a0,
      (1 - alpha) / a0,
    )
  }

  static peaking(fc: number, fs: number, q: number, gainDb: number): BiquadFilter {
    const A = Math.pow(10, gainDb / 40)
    const w0 = 2 * Math.PI * fc / fs
    const alpha = Math.sin(w0) / (2 * q)
    const cos_w0 = Math.cos(w0)
    const a0 = 1 + alpha / A
    return new BiquadFilter(
      (1 + alpha * A) / a0,
      -2 * cos_w0 / a0,
      (1 - alpha * A) / a0,
      -2 * cos_w0 / a0,
      (1 - alpha / A) / a0,
    )
  }
}
```

```ts
// src/core/shared/interpolate.ts
export function catmullRom(p0: number, p1: number, p2: number, p3: number, t: number): number {
  const t2 = t * t, t3 = t2 * t
  return 0.5 * (
    (2 * p1) +
    (-p0 + p2) * t +
    (2 * p0 - 5 * p1 + 4 * p2 - p3) * t2 +
    (-p0 + 3 * p1 - 3 * p2 + p3) * t3
  )
}

export function lttbDecimate(data: Float32Array, threshold: number): Float32Array {
  const len = data.length
  if (threshold >= len || threshold <= 2) return data

  const sampled = new Float32Array(threshold)
  sampled[0] = data[0]
  sampled[threshold - 1] = data[len - 1]

  const every = (len - 2) / (threshold - 2)
  let a = 0

  for (let i = 1; i < threshold - 1; i++) {
    const rangeStart = Math.floor((i - 1) * every) + 1
    const rangeEnd = Math.min(Math.floor(i * every) + 1, len)
    const nextStart = Math.floor(i * every) + 1
    const nextEnd = Math.min(Math.floor((i + 1) * every) + 1, len)

    let avgX = 0, avgY = 0, count = 0
    for (let j = nextStart; j < nextEnd; j++) {
      avgX += j; avgY += data[j]; count++
    }
    avgX /= count; avgY /= count

    let maxArea = -1, maxIdx = rangeStart
    for (let j = rangeStart; j < rangeEnd; j++) {
      const area = Math.abs((a - avgX) * (data[j] - data[a]) - (a - j) * (avgY - data[a]))
      if (area > maxArea) { maxArea = area; maxIdx = j }
    }
    sampled[i] = data[maxIdx]
    a = maxIdx
  }
  return sampled
}
```

```ts
// src/core/pid/filters.ts
import { BiquadFilter } from '@core/shared/biquad'
import type { FilterConfig } from './types'

export class PT1Filter {
  private alpha: number
  private state = 0

  constructor(alpha: number) { this.alpha = alpha }

  static fromCutoff(fc: number, fs: number): PT1Filter {
    const rc = 1 / (2 * Math.PI * fc)
    const dt = 1 / fs
    return new PT1Filter(dt / (rc + dt))
  }

  process(x: number): number {
    this.state += this.alpha * (x - this.state)
    return this.state
  }

  reset() { this.state = 0 }
}

export interface FilterBank {
  gyro: PT1Filter
  dterm: PT1Filter
  notch: BiquadFilter | null
}

export function createFilterBank(config: FilterConfig, fs: number): FilterBank {
  return {
    gyro: PT1Filter.fromCutoff(config.gyroLowpassHz, fs),
    dterm: PT1Filter.fromCutoff(config.dtermLowpassHz, fs),
    notch: config.notch
      ? BiquadFilter.notch(config.notch.centerHz, fs, config.notch.q)
      : null,
  }
}
```

```ts
// src/core/pid/plant.ts
import { BiquadFilter } from '@core/shared/biquad'
import { clamp } from '@core/shared/math'
import type { PlantModel } from './types'

export interface PlantState {
  omega: number
  tauApplied: number
  resonanceFilter: BiquadFilter | null
}

export function createPlantState(plant?: PlantModel): PlantState {
  return {
    omega: 0,
    tauApplied: 0,
    resonanceFilter: plant?.resonance
      ? BiquadFilter.peaking(plant.resonance.freqHz, 8000, plant.resonance.q, plant.resonance.gain)
      : null,
  }
}

export function stepPlant(
  state: PlantState, plant: PlantModel,
  command: number, disturbanceTorque: number, dt: number,
): void {
  const u = clamp(command, -1, 1)
  let tauCmd = plant.maxTorqueNm * u

  if (plant.prop) {
    const rpmNorm = Math.abs(u)
    tauCmd = Math.sign(u) * plant.prop.thrustCoeff * rpmNorm * rpmNorm * plant.prop.rpmMax * plant.prop.rpmMax
  }

  const tauM = plant.motorTimeConstantMs / 1000
  state.tauApplied += (tauCmd - state.tauApplied) * (dt / tauM)

  let torque = state.tauApplied
  if (state.resonanceFilter) {
    torque = state.resonanceFilter.process(torque)
  }

  const dOmega = (torque - plant.dragCoeff * state.omega + disturbanceTorque) / plant.inertiaKgM2
  state.omega += dOmega * dt
}
```

```ts
// src/core/pid/controller.ts
import { clamp } from '@core/shared/math'
import { PT1Filter } from './filters'
import type { ControllerConfig } from './types'

export interface ControllerState {
  integral: number
  prevGyro: number
  prevSetpoint: number
  dtermFilter: PT1Filter
}

export function createControllerState(config: ControllerConfig): ControllerState {
  return {
    integral: 0,
    prevGyro: 0,
    prevSetpoint: 0,
    dtermFilter: PT1Filter.fromCutoff(config.filters.dtermLowpassHz, config.loopRateHz),
  }
}

export interface ControllerOutput {
  output: number
  saturated: boolean
  pTerm: number
  iTerm: number
  dTerm: number
  ffTerm: number
}

export function stepController(
  state: ControllerState, config: ControllerConfig,
  setpoint: number, gyroFiltered: number, dt: number,
): ControllerOutput {
  const { gains } = config
  const error = setpoint - gyroFiltered

  const pTerm = gains.kp * error

  const setpointRate = Math.abs(setpoint - state.prevSetpoint) / dt
  const relaxFactor = config.iTermRelax ? Math.max(0, 1 - setpointRate / 2000) : 1
  state.integral += gains.ki * error * dt * relaxFactor
  state.integral = clamp(state.integral, -config.iTermLimitNm, config.iTermLimitNm)
  const iTerm = state.integral

  const rawD = -(gyroFiltered - state.prevGyro) / dt
  let dTerm = gains.kd * state.dtermFilter.process(rawD)

  if (gains.dMin !== undefined && gains.dMax !== undefined) {
    const absD = Math.abs(dTerm)
    const factor = clamp(absD / (gains.dMax || 1), 0, 1)
    const dynKd = gains.dMin + (gains.kd - gains.dMin) * factor
    dTerm = dynKd * state.dtermFilter.process(rawD)
  }

  const ffTerm = gains.kff * (setpoint - state.prevSetpoint) / dt

  let raw = pTerm + iTerm + dTerm + ffTerm

  if (config.tpa) {
    const throttle = 0.5
    const attenuation = throttle > config.tpa.breakpoint
      ? 1 - (throttle - config.tpa.breakpoint) * config.tpa.rate / (1 - config.tpa.breakpoint)
      : 1
    raw *= clamp(attenuation, 0, 1)
  }

  const output = clamp(raw, -1, 1)
  const saturated = raw !== output

  state.prevGyro = gyroFiltered
  state.prevSetpoint = setpoint

  return { output, saturated, pTerm, iTerm, dTerm, ffTerm }
}
```

```ts
// src/core/pid/setpoint.ts
import { clamp } from '@core/shared/math'
import { radToDeg } from '@core/shared/units'
import type { SetpointProfile } from './types'

export function generateSetpoint(profile: SetpointProfile, tMs: number): number {
  switch (profile.kind) {
    case 'step':
      return tMs >= profile.startMs ? profile.amplitudeDegS : 0
    case 'ramp': {
      if (tMs < profile.startMs) return 0
      const elapsed = tMs - profile.startMs
      return clamp(elapsed / profile.durationMs, 0, 1) * profile.amplitudeDegS
    }
    case 'sine':
      return profile.amplitudeDegS * Math.sin(2 * Math.PI * profile.frequencyHz * tMs / 1000)
    case 'trace': {
      const idx = (tMs / 1000) * profile.sampleRateHz
      const i = Math.floor(idx)
      if (i >= profile.samplesDegS.length - 1) return profile.samplesDegS[profile.samplesDegS.length - 1]
      const frac = idx - i
      return profile.samplesDegS[i] * (1 - frac) + profile.samplesDegS[i + 1] * frac
    }
  }
}
```

```ts
// src/core/pid/metrics.ts
import type { SimSample, SimMetrics } from './types'

export function computeMetrics(samples: SimSample[], setpointAmplitude: number): SimMetrics {
  if (samples.length === 0 || setpointAmplitude === 0) {
    return { riseTimeMs: null, overshootPct: null, settlingTimeMs: null, steadyStateErrorDegS: null, oscillationHz: null, motorActivityRms: 0 }
  }

  const target = setpointAmplitude
  let riseTimeMs: number | null = null
  let peak = 0
  let settlingTimeMs: number | null = null
  const band = Math.abs(target) * 0.02

  for (const s of samples) {
    if (riseTimeMs === null && Math.abs(s.gyroDegS) >= Math.abs(target) * 0.9) {
      riseTimeMs = s.tMs
    }
    if (Math.abs(s.gyroDegS) > Math.abs(peak)) peak = s.gyroDegS
  }

  for (let i = samples.length - 1; i >= 0; i--) {
    if (Math.abs(samples[i].gyroDegS - target) > band) {
      settlingTimeMs = samples[i].tMs
      break
    }
  }

  const overshootPct = target !== 0 ? ((peak - target) / target) * 100 : null

  const last50 = samples.slice(-Math.floor(samples.length * 0.1))
  const avgLast = last50.reduce((s, v) => s + v.gyroDegS, 0) / last50.length
  const steadyStateErrorDegS = target - avgLast

  let zeroCrossings = 0
  for (let i = 1; i < samples.length; i++) {
    if ((samples[i].errorDegS > 0) !== (samples[i - 1].errorDegS > 0)) zeroCrossings++
  }
  const duration = (samples[samples.length - 1].tMs - samples[0].tMs) / 1000
  const oscillationHz = duration > 0 ? zeroCrossings / (2 * duration) : null

  const motorArr = new Float32Array(samples.map(s => s.motorOutput))
  let motorSum = 0
  for (let i = 0; i < motorArr.length; i++) motorSum += motorArr[i] * motorArr[i]
  const motorActivityRms = Math.sqrt(motorSum / motorArr.length)

  return { riseTimeMs, overshootPct, settlingTimeMs, steadyStateErrorDegS, oscillationHz, motorActivityRms }
}
```

```ts
// src/core/pid/simulate.ts
import type { SimConfig, SimSample, SimResult } from './types'
import { createPlantState, stepPlant } from './plant'
import { createControllerState, stepController } from './controller'
import { createFilterBank } from './filters'
import { generateSetpoint } from './setpoint'
import { computeMetrics } from './metrics'
import { mulberry32, gaussianNoise } from '@core/shared/prng'
import { degToRad, radToDeg } from '@core/shared/units'

export function simulate(config: SimConfig): SimResult {
  const { controller: ctrlCfg, plant: plantModel, noise, setpoint: sp, disturbances, durationMs } = config
  const fs = ctrlCfg.loopRateHz
  const dt = 1 / fs
  const steps = Math.floor((durationMs / 1000) * fs)

  const plantState = createPlantState(plantModel)
  const ctrlState = createControllerState(ctrlCfg)
  const filterBank = createFilterBank(ctrlCfg.filters, fs)
  const rng = mulberry32(noise.seed)

  const samples: SimSample[] = []
  const decimation = Math.max(1, Math.floor(fs / 1000))

  for (let i = 0; i < steps; i++) {
    const tMs = (i / fs) * 1000
    const setpointDegS = generateSetpoint(sp, tMs)
    const setpointRad = degToRad(setpointDegS)

    let distTorque = 0
    for (const d of disturbances) {
      if (tMs >= d.startMs && tMs < d.startMs + d.durationMs) {
        distTorque += d.torqueNm
      } else if (d.kind === 'step' && tMs >= d.startMs) {
        distTorque += d.torqueNm
      }
    }

    const gyroRad = plantState.omega
    const gyroDegS = radToDeg(gyroRad)

    let noiseDeg = 0
    if (noise.kind === 'gaussian' && noise.gaussianStdDegS) {
      noiseDeg = gaussianNoise(rng) * noise.gaussianStdDegS
    }

    let measured = gyroRad + degToRad(noiseDeg)
    measured = filterBank.gyro.process(measured)
    if (filterBank.notch) measured = filterBank.notch.process(measured)
    const measuredDegS = radToDeg(measured)

    const ctrl = stepController(ctrlState, ctrlCfg, setpointDegS, measuredDegS, dt)

    stepPlant(plantState, plantModel, ctrl.output, distTorque, dt)

    if (i % decimation === 0) {
      samples.push({
        tMs,
        setpointDegS,
        gyroDegS,
        gyroMeasuredDegS: measuredDegS,
        errorDegS: setpointDegS - measuredDegS,
        pTerm: ctrl.pTerm,
        iTerm: ctrl.iTerm,
        dTerm: ctrl.dTerm,
        ffTerm: ctrl.ffTerm,
        motorOutput: ctrl.output,
        saturated: ctrl.saturated,
      })
    }
  }

  const amplitude = sp.kind === 'step' ? sp.amplitudeDegS
    : sp.kind === 'ramp' ? sp.amplitudeDegS
    : sp.kind === 'sine' ? sp.amplitudeDegS
    : 0
  const metrics = computeMetrics(samples, amplitude)

  return { samples, metrics }
}
```
